# Questão 07

int i=3, j=5;

int *p, *q; 

p = &i;

q = &j;

 Qual é o valor das seguintes expressões? 
 a) p == &i;
 
 b) *p - *q;
 
 c)*&p;
 
 d)3* - *p/(*q)+7;

	A) Return 1; ( pois é verdade)
	B) Conteudo de i - conteudo de j = -2;(3-5)
	C) Return &i; (endereço de i)
	D)  6;
